cmake_minimum_required(VERSION 3.12)
project(Raylib_Project)

set(CMAKE_CXX_STANDARD 17)

# Raylib'in gereksiz örneklerini ve testlerini derlemeyerek hız kazanıyoruz
set(BUILD_EXAMPLES OFF CACHE BOOL "" FORCE)
set(BUILD_TESTS OFF CACHE BOOL "" FORCE)

# 1. DIŞ BAĞIMLILIK: Raylib kütüphanesini GitHub'dan çekiyoruz
include(FetchContent)
FetchContent_Declare(
  raylib
  GIT_REPOSITORY https://github.com/raysan5/raylib
  GIT_TAG        master
)
FetchContent_MakeAvailable(raylib)

# =====================================================================
# 2. ZEKİ FİLTRELEME SİSTEMİ (İlhan'ın Laboratuvar Yasası)
# =====================================================================

file(GLOB ALL_SRCS CONFIGURE_DEPENDS "*.cpp")
file(GLOB MAIN_SRCS CONFIGURE_DEPENDS "*_main.cpp")

# Yardımcıları Ayır (Düz .cpp dosyaları yardımcı nesne/kütüphane kodudur)
set(HELPER_SRCS ${ALL_SRCS})
if(MAIN_SRCS)
    list(REMOVE_ITEM HELPER_SRCS ${MAIN_SRCS})
endif()

# 3. OTOMATİK ÜRETİM DÖNGÜSÜ: Her bir _main.cpp için ayrı oyun/program üret
foreach(MAIN_FILE ${MAIN_SRCS})
    # Dosya adını al (örn: sahne_main.cpp -> sahne_main)
    get_filename_component(BASE_NAME ${MAIN_FILE} NAME_WE)
    # '_main' kısmını kırp (sahne_main -> sahne)
    string(REPLACE "_main" "" TARGET_NAME ${BASE_NAME})

    # Programı oluştur (Asıl dosya + Varsa yanındaki yardımcı kodlar)
    add_executable(${TARGET_NAME} ${MAIN_FILE} ${HELPER_SRCS})
    
    # Her bir programa RAYLIB kütüphanesini bağlıyoruz!
    target_link_libraries(${TARGET_NAME} PRIVATE raylib) 
endforeach()
